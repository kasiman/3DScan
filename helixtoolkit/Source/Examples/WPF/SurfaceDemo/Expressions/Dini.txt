u=(u-0.5)*8;
v=(v-0.5)*4*pi;

double aa = w; // 0.6;
double psi = aa;  

if (psi < 0.001) psi = 0.001;
if (psi > 0.999) psi = 0.999;
psi = psi * pi;

double sinpsi = sin(psi);
double cospsi = cos(psi);
double g = (u - cospsi * v) / sinpsi;
double s = Math.Exp(g);
double r = (2 * sinpsi) / (s + 1 / s);
double t = r * (s - 1 / s) * 0.5;
  
x = (u - t);
y = (r * cos(v));
z = (r * sin(v));
