u *= 2 * pi;
v = (v - 0.5) * 2 * pi;

double cosv = cos(v);
double sinv = sin(v);

x = cos(u) * (4 + 3.8 * cosv);
y = sin(u) * (4 + 3.8 * cosv);
z = (cosv + sinv - 1) * (1 + sinv) * Math.Log(1 - pi * v / 10) + 7.5 * sinv;
